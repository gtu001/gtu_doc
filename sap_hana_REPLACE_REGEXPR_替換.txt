 REPLACE_REGEXPR(<pattern> IN <column> WITH <replace_to> OCCURRENCE ALL)

 Ex : 
 	REPLACE_REGEXPR('^0+' IN policyNumber WITH '' OCCURRENCE ALL)

 	SELECT REPLACE_REGEXPR('([[:digit:]]{4})-([[:digit:]]{2})-([[:digit:]]{2})' IN '2014-04-01' 
WITH '\3/\2/\1' OCCURRENCE ALL) "replace_regexpr" FROM DUMMY;
